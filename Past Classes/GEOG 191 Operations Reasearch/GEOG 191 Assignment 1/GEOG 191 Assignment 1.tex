\section*{Exercise 6}
\begin{problem}
    What are the functional values for $x$ equal to $5$, $10$, and $27$ when $f(x) = 2x^3 - 7x + 1$?
\end{problem}

\begin{align*}
    f(5) &= 2(5)^3 - 7(5) + 1 \\
        &= 250 - 35 + 1 \\
        &= 216.
\end{align*}

\begin{align*}
    f(10) &= 2(10)^3 - 7(10) + 1 \\
        &= 2000 - 70 + 1 \\
        &= 1931.
\end{align*}

\begin{align*}
    f(27) &= 2(27)^3 - 7(27) + 1 \\
        &= 39266 - 289 + 1 \\
        &= 38978.
\end{align*}
